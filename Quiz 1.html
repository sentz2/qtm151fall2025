list_countries = ["data_quiz_1/korea_rep_co2_emissions.csv", "data_quiz_1/india_co2_emissions.csv"]
list_emissions = []
for country in list_countries:
    data_country = pd.read_csv(country)
    total_emissions = data_country["total_emissions"]
    mean_emissions = np.mean(total_emissions)
    list_emissions.append(mean_emissions)
print(list_emissions)
